创建 `src/store/chat.ts`（Zustand chatStore）：

内置两个 DeepSeek Provider 配置（作为 providerStore 的初始值来源，chat.ts 不管理配置，只调用 ProviderRegistry）：

```typescript
import { create } from 'zustand';
import type { SelectionContext } from '../adapters/DocumentAdapter';
import { ProviderRegistry } from '../providers/registry';
import { OpenAICompatibleLLM } from '../providers/openai-compat';
import { setupVisibilityAbort } from '../providers/queue';
import { calcCostCny } from '../providers/pricing';
import { useProviderStore } from './providers';  // 获取默认 LLM 配置

interface Message {
  id: string;
  role: 'user' | 'assistant' | 'error';
  content: string;
  isStreaming?: boolean;
  tokenCount?: number;
  costCny?: number | null;
  errorCode?: string;
  retryPrompt?: string;
}

interface ChatState {
  messages: Message[];
  isStreaming: boolean;
  abortController: AbortController | null;
  sendMessage(prompt: string, selectionCtx?: SelectionContext): Promise<void>;
  stopStreaming(): void;
  retryMessage(messageId: string): Promise<void>;
  clearHistory(): void;
}

export const useChatStore = create<ChatState>((set, get) => ({
  messages: [],
  isStreaming: false,
  abortController: null,

  async sendMessage(prompt, selectionCtx) {
    if (get().isStreaming) return; // 防止并发发送

    // 1. 构建用户消息（附带选区上下文 D-15）
    const userContent = selectionCtx && selectionCtx.kind !== 'none'
      ? `[上下文: ${JSON.stringify(selectionCtx)}]\n\n${prompt}`
      : prompt;

    const userMsg: Message = { id: crypto.randomUUID(), role: 'user', content: prompt };
    const assistantMsg: Message = { id: crypto.randomUUID(), role: 'assistant', content: '', isStreaming: true };

    set((s) => ({ messages: [...s.messages, userMsg, assistantMsg], isStreaming: true }));

    // 2. 创建 AbortController，注册 visibilitychange abort
    const controller = new AbortController();
    const cleanup = setupVisibilityAbort(controller);
    set({ abortController: controller });

    try {
      // 3. 从 providerStore 获取当前默认 LLM 配置
      const providerState = useProviderStore.getState();
      const config = ProviderRegistry.resolve('chat', () => {
        const p = providerState.providers.find((p) => p.id === providerState.defaultLLMProviderId);
        if (!p) throw new Error('默认 Provider 未配置');
        return p;
      });

      // 4. 构建消息历史（系统 prompt + 历史 + 当前）
      const messages = [
        { role: 'system' as const, content: 'You are Aster, an AI assistant for Microsoft Office.' },
        ...get().messages
          .filter((m) => m.role !== 'error' && !m.isStreaming)
          .map((m) => ({ role: m.role as 'user' | 'assistant', content: m.content })),
        { role: 'user' as const, content: userContent },
      ];

      // 5. 流式生成
      const llm = new OpenAICompatibleLLM();
      for await (const event of llm.streamChat(messages, config as import('../providers/types').LLMConfig, controller.signal)) {
        if (event.type === 'delta') {
          set((s) => ({
            messages: s.messages.map((m) =>
              m.id === assistantMsg.id ? { ...m, content: m.content + event.content } : m,
            ),
          }));
        } else if (event.type === 'usage') {
          const costCny = calcCostCny(
            { promptTokens: event.promptTokens, completionTokens: event.completionTokens },
            config.providerId,
          );
          set((s) => ({
            messages: s.messages.map((m) =>
              m.id === assistantMsg.id
                ? { ...m, tokenCount: event.totalTokens, costCny }
                : m,
            ),
          }));
        }
      }
    } catch (e) {
      if (e instanceof Error && e.name === 'AbortError') {
        // 用户停止——保留已生成内容，标记非流式状态
        set((s) => ({
          messages: s.messages.map((m) =>
            m.id === assistantMsg.id ? { ...m, isStreaming: false } : m,
          ),
        }));
        return;
      }
      // 错误——替换 assistant 气泡为 error 气泡（D-10/D-11）
      const errCode = (e as Record<string, string>)?.code ?? 'NETWORK';
      set((s) => ({
        messages: s.messages.map((m) =>
          m.id === assistantMsg.id
            ? { ...m, role: 'error', content: (e as Error).message, errorCode: errCode, retryPrompt: prompt, isStreaming: false }
            : m,
        ),
      }));
    } finally {
      cleanup();
      set({ isStreaming: false, abortController: null });
      // 标记最后一条消息非流式
      set((s) => ({
        messages: s.messages.map((m) =>
          m.id === assistantMsg.id ? { ...m, isStreaming: false } : m,
        ),
      }));
    }
  },

  stopStreaming() {
    get().abortController?.abort();
  },

  async retryMessage(messageId) {
    const msg = get().messages.find((m) => m.id === messageId);
    if (!msg?.retryPrompt) return;
    // 移除失败气泡，重新发送
    set((s) => ({ messages: s.messages.filter((m) => m.id !== messageId) }));
    await get().sendMessage(msg.retryPrompt);
  },

  clearHistory() {
    set({ messages: [], isStreaming: false, abortController: null });
  },
}));

// Named selector hooks（性能：避免全 re-render）
export const useMessages = () => useChatStore((s) => s.messages);
export const useIsStreaming = () => useChatStore((s) => s.isStreaming);
```

**重要（D-15 / B4）：** `autoAttach: boolean` 及其 `setAutoAttach` action 在此 Wave 3 实现，Wave 4 的 SelectionPill 直接消费。初始值从 `storage.get(STORAGE_KEYS.SELECTION_AUTO_ATTACH) ?? true` 读取，setAutoAttach 写回 storage。

```typescript
// 内置 Provider（isBuiltIn=true，不可删除）
const BUILT_IN_PROVIDERS: ProviderConfig[] = [
  { id: 'deepseek', name: 'DeepSeek', baseURL: 'https://api.deepseek.com', model: 'deepseek-v4-flash', isBuiltIn: true },
  { id: 'aihubmix', name: 'AiHubMix', baseURL: 'https://api.aihubmix.com/v1', model: 'gpt-image-1', isBuiltIn: true },
];

interface ProviderState {
  providers: ProviderConfig[];
  defaultLLMProviderId: string;
  autoAttach: boolean;        // D-15：选区自动附带开关，持久化到 SELECTION_AUTO_ATTACH
  addProvider(config: Omit<ProviderConfig, 'id'>): void;
  updateProvider(id: string, patch: Partial<ProviderConfig>): void;
  removeProvider(id: string): void;
  setDefaultLLM(id: string): void;
  setKey(providerId: string, apiKey: string): void;
  getKey(providerId: string): string | null;
  setAutoAttach(v: boolean): void;  // 写 storage，供 SettingsPanel 和 SelectionPill 调用
}

export const useProviderStore = create<ProviderState>((set, get) => ({
  providers: BUILT_IN_PROVIDERS,
  defaultLLMProviderId: 'deepseek',
  autoAttach: storage.get<boolean>(STORAGE_KEYS.SELECTION_AUTO_ATTACH) ?? true,

  addProvider(config) {
    const id = crypto.randomUUID();
    const newProvider = { ...config, id };
    const updated = [...get().providers, newProvider];
    set({ providers: updated });
    storage.set(STORAGE_KEYS.PROVIDERS, updated);
  },

  updateProvider(id, patch) {
    const updated = get().providers.map((p) => (p.id === id ? { ...p, ...patch } : p));
    set({ providers: updated });
    storage.set(STORAGE_KEYS.PROVIDERS, updated);
  },

  removeProvider(id) {
    const p = get().providers.find((p) => p.id === id);
    if (p?.isBuiltIn) return; // 内置 Provider 不可删除
    const updated = get().providers.filter((p) => p.id !== id);
    set({ providers: updated });
    storage.set(STORAGE_KEYS.PROVIDERS, updated);
  },

  setDefaultLLM(id) {
    set({ defaultLLMProviderId: id });
    storage.set(STORAGE_KEYS.DEFAULT_PROVIDER, id);
  },

  setKey(providerId, apiKey) {
    storage.set(`aster:keys:${providerId}`, apiKey);
  },

  getKey(providerId) {
    return storage.get<string>(`aster:keys:${providerId}`);
  },

  setAutoAttach(v) {
    set({ autoAttach: v });
    storage.set(STORAGE_KEYS.SELECTION_AUTO_ATTACH, v);
  },
}));

// 持久化逻辑：读时调用 hydrateFromStorage()（在 main.tsx Office.onReady 后调用）
export function hydrateFromStorage(): void {
  const stored = storage.get<ProviderConfig[]>(STORAGE_KEYS.PROVIDERS);
  const defaultId = storage.get<string>(STORAGE_KEYS.DEFAULT_PROVIDER) ?? 'deepseek';
  const autoAttach = storage.get<boolean>(STORAGE_KEYS.SELECTION_AUTO_ATTACH) ?? true;
  if (stored && stored.length > 0) {
    useProviderStore.setState({ providers: stored, defaultLLMProviderId: defaultId, autoAttach });
  }
}
```

注意：`hydrateFromStorage` 必须在 `main.tsx` 的 `Office.onReady` 回调内调用，在 `root.render` 之前（按 Phase 1 main.tsx 模式）。

```typescript
async insert(content: InsertableContent): Promise<void> {
  if (content.type !== 'text') {
    throw new UnsupportedOperationError(`PPT Phase 2 仅支持 text 写回，${content.type} 在 Phase 4 实现`);
  }
  try {
    await PowerPoint.run(async (ctx) => {
      const slides = ctx.presentation.getSelectedSlides();
      const slide = slides.getItemAt(0);
      const shapes = slide.shapes;
      shapes.load('items');
      await ctx.sync();
      if (shapes.items.length > 0) {
        // 写入第一个文本框（覆盖模式，Phase 4 支持更精确的目标选择）
        shapes.items[0].textFrame.text = content.value;
      }
      await ctx.sync();
    });
  } catch (err) {
    if (err instanceof UnsupportedOperationError) throw err;
    throw new HostApiError('PPT text 写回失败', err);
  }
}
```

```typescript
      // replace 模式：替换选区（或在光标处插入，Word API 同一调用）
      sel.insertText(content.value, Word.InsertLocation.replace);
      await ctx.sync();
    });
  } catch (err) {
    if (err instanceof UnsupportedOperationError) throw err;
    throw new HostApiError('Word text 写回失败', err);
  }
}
```
