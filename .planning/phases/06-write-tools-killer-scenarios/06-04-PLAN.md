function normalizeText(s: string): string {
  return s.replace(/\r\n/g, '\n').trimEnd();

// deleteParagraphByContent（lines 168-191）— inverse Record 签名范式：
async deleteParagraphByContent(args: Record<string, unknown>): Promise<void> {
  const text = args.text as string;  // ← 内部解构，不是位置参
  await Word.run(async (ctx) => {
    const paras = ctx.document.body.paragraphs;
    paras.load('items/text');
    await ctx.sync();
    const normalTarget = normalizeText(text);
    for (let i = paras.items.length - 1; i >= 0; i--) {
      if (normalizeText(paras.items[i].text) === normalTarget) {
        paras.items[i].delete();
        await ctx.sync();
        return;
      }
    }
    throw new HostApiError('Word deleteParagraphByContent: 目标段落已不存在', undefined);
  });
}

// replaceParagraphAt before-image + D-11 expected_state 范式：
async replaceParagraphAt(index: number, newText: string, expectedText?: string): Promise<{ beforeImage: string }> {
  return await Word.run(async (ctx) => {
    const paras = ctx.document.body.paragraphs;
    paras.load('items/text');
    await ctx.sync();
    if (index < 0 || index >= paras.items.length) throw new HostApiError(`段落 index=${index} 不存在`);
    const currentText = normalizeText(paras.items[index].text);
    if (expectedText && normalizeText(expectedText) !== currentText) throw new HostApiError('并发修改冲突');
    const beforeImage = paras.items[index].text as string;
    paras.items[index].insertText(newText, Word.InsertLocation.replace);
    await ctx.sync();
    return { beforeImage };
  });
}
