```typescript
const HUMAN_LABEL_TEXT_CAP = 30;
export const appendParagraph: ToolDef<AppendParagraphArgs> = {
  name: 'append_paragraph', kind: 'write',
  humanLabel: ({ text }) =>
    `在文档末尾追加段落「${text.slice(0, HUMAN_LABEL_TEXT_CAP)}${text.length > HUMAN_LABEL_TEXT_CAP ? '…' : ''}」`,
  async execute({ text }, ctx): Promise<ToolResult> {
    await (ctx.adapter as WordAdapter).appendParagraph(text);
    const reverse: ReverseDescriptor = { tool: 'delete_paragraph_by_content', args: { text } };
    const postState = { kind: 'word_paragraph' as const, content: text };
    return { ok: true, data: { written: text.length }, reverse, postState };
  },
};
```

<!-- Source: PATTERNS.md — replace_paragraph reverse 结构（lines 201-216）-->
```typescript
// replace_paragraph reverse（before-image 模式）：
const reverse: ReverseDescriptor = {
  tool: 'restore_paragraph_at',
  args: { index, expectedText: newText, restoreText: beforeImage }  // Record 对象
};
// D-11 expected_state 支持（replace_paragraph 是高风险写）

// replace_selection inverse（降级模式）：
// reverse.tool = 'delete_paragraph_by_content'（近似 inverse，用写入内容指纹）
// reverse.args = { text: newText }（Record 对象）
// 理由：replace_selection 无法精确定位原始段落 index（选区位置不稳定）；
// 用写入的新文本做指纹来"近似还原"——如果还原失败，replay engine 标 skipped_error，
// 用户看到「无法自动回滚此步」提示。不使用 'noop_inverse' 因为
// 'noop_inverse' 不在 executeReverse switch 中，会触发 default: throw
// → 被 replayUndoStep 的 catch 捕获 → 标 skipped_error（等效）。
// 选 delete_paragraph_by_content 近似 inverse 略优于 noop：至少有概率成功回滚。
```
