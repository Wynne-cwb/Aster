```typescript
 * ⚠️ 真机 UAT：slides.add 后同 run 内几何形状 + fill/font 在 Office for Web 稳定性（addTextBox 已证）。
 * ⚠️ Record/位置参不涉及（本方法是正向，收 ShapeSpec[]；inverse 是已有 deleteSlideByIndex 收 Record）。
 */
async applySlideLayout(
  shapeSpecs: Array<{
    shapeType: string; rect: { left:number; top:number; width:number; height:number };
    text?: string; font?: { size?:number; bold?:boolean; color?:string; name?:string };
    fillColor?: string; lineColor?: string; lineWeight?: number; align?: string;
  }>,
): Promise<{ capturedIndex: number; capturedId: string; slideIndex: number; newShapeIds: string[] }> {
  try {
    return await PowerPoint.run(async (ctx) => {
      const slides = ctx.presentation.slides;
      slides.load('items'); await ctx.sync();           // sync 1
      slides.add();                                     // 追加新页到末尾
      slides.load('items'); await ctx.sync();           // sync 2: reload
      const sorted = [...(slides.items as Array<{ index:number; id:string; load:(f:string[])=>void; shapes: { addTextBox:(t:string,o:object)=>{load:(f:string[])=>void;id:string}; addGeometricShape:(t:string,o:object)=>{load:(f:string[])=>void;id:string;type:string; textFrame:{textRange:{text:string; font:Record<string,unknown>; paragraphFormat:{horizontalAlignment:string}}}; fill:{setSolidColor:(c:string)=>void}; lineFormat:{color:string;weight:number;visible:boolean} } } }>)]
        .sort((a,b)=>a.index-b.index);
      if (sorted.length === 0) throw new HostApiError('PPT applySlideLayout: 建页后列表为空', undefined);
      const newSlide = sorted[sorted.length - 1];
      newSlide.load(['id','index']);                    // 捕双定位指纹
      // 逐 spec 建形状（TextBox 用 addTextBox；其余几何用 addGeometricShape）
      const created: Array<{ load:(f:string[])=>void; id:string }> = [];
      for (const s of shapeSpecs) {
        if (s.shapeType === 'TextBox') {
          const tb = newSlide.shapes.addTextBox(s.text ?? '', s.rect);
          // font/align 写在 sync 后或同闭包内（TextBox.textFrame）—参 addShape/ setShapeTextFont 范式
          created.push(tb);
        } else {
          const gs = newSlide.shapes.addGeometricShape(s.shapeType, s.rect);
          if (s.fillColor) gs.fill.setSolidColor(s.fillColor);
          if (s.lineColor) { gs.lineFormat.color = s.lineColor; gs.lineFormat.visible = true; }
          if (s.lineWeight !== undefined) gs.lineFormat.weight = s.lineWeight;
          if (s.text !== undefined) gs.textFrame.textRange.text = s.text;   // 几何形状文字（白名单类型）
          created.push(gs);
        }
      }
      await ctx.sync();                                 // sync 3: 形状创建 + 基本属性
      // 设字体/对齐（需 textFrame；TextBox 与含文字几何形状统一处理）+ 收 id
      for (let i=0;i<shapeSpecs.length;i++) {
        const s = shapeSpecs[i]; const sh = created[i] as unknown as {
          load:(f:string[])=>void; id:string;
          textFrame?: { textRange: { font: Record<string,unknown>; paragraphFormat: { horizontalAlignment: string } } };
        };
        if (sh.textFrame && s.font) { const f = sh.textFrame.textRange.font;
          if (s.font.size!==undefined) f.size = s.font.size as never;
          if (s.font.bold!==undefined) f.bold = s.font.bold as never;
          if (s.font.color!==undefined) f.color = s.font.color as never;
          if (s.font.name!==undefined) f.name = s.font.name as never; }
        if (sh.textFrame && s.align) sh.textFrame.textRange.paragraphFormat.horizontalAlignment = normalizeAlignment(s.align);
        sh.load(['id']);
      }
      await ctx.sync();                                 // sync 4: 字体/对齐 + load id
      const newShapeIds = created.map((c) => (c as unknown as {id:string}).id);
      return { capturedIndex: newSlide.index, capturedId: newSlide.id, slideIndex: newSlide.index + 1, newShapeIds };
    });
  } catch (err) {
    if (err instanceof HostApiError) throw err;
    throw new HostApiError('PPT applySlideLayout 失败', err);
  }
}
```
