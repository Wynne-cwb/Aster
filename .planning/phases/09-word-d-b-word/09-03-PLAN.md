```typescript
case 'selection_detail': {
  try {
    return await Word.run(async (ctx) => {
      const selection = ctx.document.getSelection();
      const body = ctx.document.body;
      const paras = body.paragraphs;

      // 运行时门控（D-02/D-03）：检查 WordApi 1.6 可用性
      const supportsUniqueId =
        typeof Office !== 'undefined' &&
        Office.context?.requirements?.isSetSupported('WordApi', '1.6') === true;

      // 加载 selection text + paragraphs（按支持情况加载 uniqueLocalId）
      const paraLoadStr = supportsUniqueId
        ? 'items/text,items/uniqueLocalId'
        : 'items/text';
      selection.load('text');
      paras.load(paraLoadStr);
      await ctx.sync();

      const text = selection.text;
      if (text.length === 0) {
        return { ok: true, data: { kind: 'none' } } satisfies ReadableResult;
      }

      const selNorm = normalizeText(text);
      let paragraphIndex = -1;
      let uniqueLocalId: string | null = null;
      let matchCount = 0;

      for (let i = 0; i < paras.items.length; i++) {
        if (normalizeText(paras.items[i].text) === selNorm) {
          matchCount++;
          if (matchCount === 1) {
            paragraphIndex = i;
            uniqueLocalId = supportsUniqueId
              ? (paras.items[i].uniqueLocalId ?? null)
              : null;
          }
          // 若有多个同名段落，继续统计（不立即退出），但取第一个 index
        }
      }

      // 如果文本指纹没有完整匹配任何段落（如 selection 是段落子集或跨段落）
      // → paragraphIndex = -1
      if (paragraphIndex === -1) {
        return {
          ok: true,
          data: {
            kind: 'word',
            charCount: text.length,
            text,
            paragraphIndex: -1,
            uniqueLocalId: null,
            selectionSpansMultipleParagraphs: true,
          },
        } satisfies ReadableResult;
      }

      return {
        ok: true,
        data: {
          kind: 'word',
          charCount: text.length,
          text,
          paragraphIndex,
          uniqueLocalId,
        },
      } satisfies ReadableResult;
    });
  } catch (err) {
    throw new HostApiError('Word selection_detail 失败', err);
  }
}
```

- `supportsUniqueId` 检查中用 `typeof Office !== 'undefined'` 防止 test 环境未 mock Office 时崩溃
- `uniqueLocalId ?? null` 处理 Desktop Word 返回 null 的情况（issue #4258）
- 不使用 `compareLocationWith`（需要额外 sync，v2.1 用文本指纹快路径，见 RESEARCH.md Q5 说明）
