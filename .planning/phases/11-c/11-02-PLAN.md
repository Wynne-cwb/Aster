executeReverse switch（line 251-306，当前）——在 'noop_inverse' case 之前插入完整的 per-subOp 手改防御实现（D-09）：
```typescript
case 'batch_reverse': {
  const ops = reverse.args.ops as Array<{ tool: string; args: Record<string, unknown> }>;
  const reversedOps = [...ops].reverse(); // 逆序：最后写的先撤（D-07/SC#3）

  // 优先路径：executeBatchReverse 单闭包（D-08 对称设计）
  if ('executeBatchReverse' in adapter &&
      typeof (adapter as Record<string, unknown>).executeBatchReverse === 'function') {
    // 单闭包逆序 undo（adapter 层内部 per-subOp try/catch，D-09）
    await (adapter as { executeBatchReverse: (ops: typeof reversedOps) => Promise<void> }).executeBatchReverse(reversedOps);
    break;
  }

  // 降级路径：per-subOp 手改防御 + continue-on-error（D-09）
  // 从 entry.subOps 中拿对应 postState 做手改检测
  // entry 通过 operationLog 查找（reverse.args.ops 与 entry.subOps 顺序一致，逆序遍历）
  let rolledBack = 0;
  let skippedManual = 0;
  let skippedError = 0;

  for (const subOp of reversedOps) {
    try {
      // 从 reverse.args 中找对应 postState（通过 subOp.args 匹配）
      // 简化：直接调 readTargetState + isTargetStateConsistent 检查是否手改
      // 若 subOp.postState 存在，读当前状态对比；否则跳过手改检测直接撤
      const subPostState = (subOp as { postState?: PostStateSnapshot }).postState;
      if (subPostState) {
        const currentState = await readTargetState(subPostState, adapter);
        if (currentState !== undefined && !isTargetStateConsistent(subPostState, currentState)) {
          // 手改过：跳过此 subOp（D-09 per-subOp 手改防御）
          skippedManual++;
          continue;
        }
      }
      await executeReverse({ tool: subOp.tool, args: subOp.args }, adapter);
      rolledBack++;
    } catch {
      // 报错：跳过此 subOp（continue-on-error，D-09）
      skippedError++;
    }
  }

  // 聚合三态结果附加到 reverse.args 供 SummaryModal 显示
  // （三态计数在 replayUndoSingle/replayUndoAll 的 UndoResult 中体现）
  // 注意：batch 整体的 rolled_back/skippedManual/skippedError 由外层 executeReverse 调用者汇总
  // 此处仅记录，不直接返回（executeReverse 返回 void）
  if (skippedManual > 0 || skippedError > 0) {
    // 将三态挂到 reverse.args 让外层知道（供 SummaryModal）
    // 用 Object.assign 避免修改原对象
    Object.assign(reverse.args, {
      _batchUndoResult: { rolledBack, skippedManual, skippedError },
    });
  }

  break;
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| executeReverse case 'batch_reverse' → adapter.executeBatchReverse | batch_reverse 的 ops 参数来自 OperationLog reverse.args（由 adapter 在执行时构建，非 LLM 直接控制）|

| Threat ID | Category | Component | Disposition | Mitigation Plan |
