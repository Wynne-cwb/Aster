**真打方式（用 Node.js fetch 或 curl）：**

建议用临时 Node.js 脚本（不用 curl，方便处理 JSON）：

```bash
# 读取 API Key（从 .env.local 手动复制或用 dotenv）
# 真打完后立即截断，不保留完整 3MB base64

node -e "
const fs = require('fs');
const key = process.env.AIHUBMIX_API_KEY || fs.readFileSync('.env.local','utf8').match(/AIHUBMIX_API_KEY=(.+)/)?.[1]?.trim();
if (!key) { console.error('找不到 AIHUBMIX_API_KEY'); process.exit(1); }
console.log('Key found:', key.slice(0, 8) + '...');

// ① doubao
fetch('https://aihubmix.com/v1/models/doubao/doubao-seedream-5.0-lite/predictions', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json', 'Authorization': 'Bearer ' + key },
  body: JSON.stringify({ input: { prompt: '一棵树', size: '2K', sequential_image_generation: 'disabled', stream: false, response_format: 'url', watermark: true } })
}).then(r => { console.log('doubao status:', r.status); return r.json(); })
  .then(j => {
    console.log('doubao output keys:', Object.keys(j));
    // 截断 URL（D-16：URL 用占位符）
    const fixture = { output: (j.output || []).map(item => ({ url: '<truncated-for-ci>' })) };
    fs.writeFileSync('src/providers/__fixtures__/doubao-response.json', JSON.stringify(fixture, null, 2));
    console.log('doubao fixture 写入完成，URL 已截断');
  }).catch(e => console.error('doubao 失败:', e.message));
"
```

```bash
# ③ gemini
node -e "
const fs = require('fs');
const key = process.env.AIHUBMIX_API_KEY || fs.readFileSync('.env.local','utf8').match(/AIHUBMIX_API_KEY=(.+)/)?.[1]?.trim();
fetch('https://aihubmix.com/gemini/v1beta/models/gemini-3.1-flash-image-preview:streamGenerateContent', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json', 'x-goog-api-key': key },
  body: JSON.stringify({ contents: [{ role: 'user', parts: [{ text: '画一棵树' }] }], generationConfig: { responseModalities: ['TEXT','IMAGE'], imageConfig: { aspectRatio: '1:1', imageSize: '1k' } } })
}).then(r => { console.log('gemini status:', r.status); return r.json(); })
  .then(chunks => {
    console.log('gemini chunks count:', chunks.length);
    // 截断 thoughtSignature + inlineData.data（D-16）
    const fixture = chunks.map(chunk => {
      if (!chunk.candidates) return chunk;
      return {
        ...chunk,
        candidates: chunk.candidates.map(cand => ({
          ...cand,
          content: cand.content ? {
            ...cand.content,
            parts: (cand.content.parts || []).map(part => {
              const p = {};
              if (part.thoughtSignature !== undefined) p.thoughtSignature = '<truncated-for-ci>';
              if (part.inlineData) p.inlineData = { data: part.inlineData.data.slice(0, 4), mimeType: part.inlineData.mimeType };
              if (part.text !== undefined) p.text = part.text;
              return p;
            })
          } : cand.content
        }))
      };
    });
    fs.writeFileSync('src/providers/__fixtures__/gemini-response.json', JSON.stringify(fixture, null, 2));
    console.log('gemini fixture 写入完成，base64 + thoughtSignature 已截断');
  }).catch(e => console.error('gemini 失败:', e.message));
"
```

**doubao CORS 开放问题处理（Pitfall 4 + Open Q 1）：**
真打脚本在 Node.js 环境运行（无 CORS 限制），可以正常拿到响应。但 aihubmix-image.ts 里 `fetchUrlToBase64` 是在浏览器侧调用的，遭遇 CORS 是运行时问题（不是本 plan 的阻塞项）。fixture 录制时顺便记录：doubao 返回的 URL 的 host 是否含 CORS 头（在 Node.js 侧用 `response.headers.get('access-control-allow-origin')` 检查签名 URL 的 CORS policy）。

**若任何一路真打失败（非网络问题）：**
- 记录在 SUMMARY.md：失败路径 + 响应状态码 + 简要 body
- fixture 保持 Plan 01 的占位结构（测试仍按 mock fetch 运行，不影响 CI）
- 不阻塞 Phase 14 完成——fixture 可以是"占位结构，真打失败原因已记录"
