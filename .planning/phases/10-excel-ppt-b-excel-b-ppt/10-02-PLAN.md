async setRangeValues(address: string, values: unknown[][]): Promise<{ beforeImage: {...} }> {
  try {
    return await Excel.run(async (ctx) => {
      const range = ctx.workbook.worksheets.getActiveWorksheet().getRange(address);
      range.load(['values', 'address']);
      await ctx.sync(); // sync 1: before-image
      const beforeImage = { address: range.address as string, values: range.values as unknown[][] };
      range.values = values;
      await ctx.sync(); // sync 2: write
      return { beforeImage };
    });
  } catch (err) { throw new HostApiError('Excel setRangeValues 失败', err); }
}

async deleteChartByName(args: Record<string, unknown>): Promise<void> {
  const chartName = args.chartName as string;
  try {
    await Excel.run(async (ctx) => {
      const sheet = ctx.workbook.worksheets.getActiveWorksheet();
      const chart = sheet.charts.getItemOrNullObject(chartName);
      chart.load('isNullObject');
      await ctx.sync();
      if (!chart.isNullObject) { chart.delete(); await ctx.sync(); }
    });
  } catch (err) { if (err instanceof HostApiError) throw err; throw new HostApiError('Excel deleteChartByName 失败', err); }
}
```

**EXCEL-01 formatExcelRange：**
- 两 sync：sync 1 load before-image（range.format.load(['fill/color', 'font/bold', 'font/color', 'font/size', 'font/name', 'horizontalAlignment']) + range.load('numberFormat')），sync 2 写入

- `restoreRangeFormat(args: Record<string, unknown>)` — 解包 address/numberFormat/fillColor 等，逐一判断非 null 才写（避免覆盖用户手动设置的其他格式）

**EXCEL-02 setColumnRowSize：**
- indices 是数组，批量 load → 单次 sync（不在循环内 sync）
- autoFit：column → `range.format.autofitColumns()`；row → `range.format.autofitRows()`
- before-image：`beforeSizes: Array<{ index: number; size: number }>` 数组（autoFit 前的真实尺寸）
- `restoreColumnRowSize(args: Record<string, unknown>)` — beforeSizes 逐一写回 columnWidth/rowHeight

**EXCEL-04 setAutoFilter：**
- before-image 只存 `{ hadFilter: boolean, address?: string }`（不存 criteria，see RESEARCH.md A6 assumption）
- 筛选框读取：`sheet.autoFilter.load(['enabled']); await ctx.sync(); const hadFilter = sheet.autoFilter.enabled as boolean`
- inverse 还原：hadFilter=true → `sheet.autoFilter.apply(address, 0, {})` 仅恢复筛选框；false → `sheet.autoFilter.remove()`
- description 标注「undo 后仅恢复筛选框，不恢复筛选条件」

**EXCEL-06 addConditionalFormat（clearAll + 重建策略）：**
- before-image：先 `range.conditionalFormats.load('items')` 读取现有格式快照（序列化 cellValue/colorScale/dataBar 三种）
- 写入：`range.conditionalFormats.add(Excel.ConditionalFormatType.xxx)` + 设置规则
- inverse `restoreConditionalFormat`：先 `range.conditionalFormats.clearAll()` → 再按 beforeFormats 重建（clearAll + 重建是幂等的安全路径，防 index 漂移）
- MVP：beforeFormats 序列化只含 cellValue/colorScale/dataBar 核心属性

**EXCEL-07 createTable：**
- `sheet.tables.add(address, hasHeaders ?? false)` → `if (tableName) table.name = tableName`

- inverse `deleteTableByName`：完全复用 deleteChartByName 范式（getItemOrNullObject + isNullObject + delete）

**EXCEL-08 freezePanes：**

export const formatExcelRangeTool: ToolDef = {
  name: 'format_excel_range',
  kind: 'write',
  description: '设置单元格格式：数字格式/字体/填充色/对齐方式',
  parameters: {
    type: 'object',
    properties: {
      address: { type: 'string', description: '单元格区域如 A1:D10' },
      numberFormat: { type: 'string', description: '数字格式字符串，如 #,##0.00' },
      fill: { type: 'object', properties: { color: { type: 'string' } } },
      font: { type: 'object', properties: { bold: { type: 'boolean' }, color: { type: 'string' }, size: { type: 'number' }, name: { type: 'string' } } },
      alignment: { type: 'string', enum: ['Left', 'Center', 'Right', 'General'] },
    },
    required: ['address'],
  },
  humanLabel: ({ address, numberFormat }: { address: string; numberFormat?: string }) =>
    `设置 ${address} 格式${numberFormat ? `（数字格式 ${numberFormat}）` : ''}`,
  async execute(args, ctx): Promise<ToolResult> {
    const { address, numberFormat, fill, font, alignment } = args as Record<string, unknown>;
    const { beforeImage } = await (ctx.adapter as ExcelAdapter).formatExcelRange(
      address as string, { numberFormat: numberFormat as string | undefined, fill: fill as { color?: string } | undefined, font: font as Record<string, unknown> | undefined, alignment: alignment as string | undefined }
    );
    const reverse: ReverseDescriptor = { tool: 'restore_range_format', args: { ...beforeImage } };
    const postState: PostStateSnapshot = { kind: 'excel_range_format', content: { address: address as string } };
    return { ok: true, data: { address }, reverse, postState };
  },
};
```
