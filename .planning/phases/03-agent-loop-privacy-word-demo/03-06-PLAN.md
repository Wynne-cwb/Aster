```typescript
import { useState } from 'react';
import { Trans } from '@lingui/react/macro';
import { useAgentStore } from '../agent/agentStore';
import type { Message } from '../store/chat';

function ToolResultCard({ message }: { message: Message }): React.ReactElement {
  const continueRun = useAgentStore((s) => s.continueRun);
  const abort = useAgentStore((s) => s.abort);
  const [expanded, setExpanded] = useState(false);

  // soft-landing 卡片：两按钮（继续 20 步 / 停下）— Plan 03 loop.ts push 时 toolName='soft-landing'
  if (message.toolName === 'soft-landing') {
    return (
      <div className="aster-tool-card aster-tool-card--soft-landing">
        <div className="aster-tool-card__title">{message.content}</div>
        <div className="aster-tool-card__actions">
          <button
            type="button"
            className="aster-iconbtn aster-iconbtn--primary"
            onClick={() => continueRun()}
          >
            <Trans>继续 20 步</Trans>
          </button>
          <button
            type="button"
            className="aster-iconbtn"
            onClick={() => abort('user')}
          >
            <Trans>停下</Trans>
          </button>
        </div>
      </div>
    );
  }

  // 常规 role='tool' 折叠卡：header 显示 humanLabel（chatStore Message.content）
  // 点开展开 toolResult JSON
  const showLabel = message.content || message.toolName || 'tool';

  return (
    <div className={`aster-tool-card${message.toolResult?.ok === false ? ' aster-tool-card--error' : ''}`}>
      <button
        type="button"
        className="aster-tool-card__header"
        onClick={() => setExpanded((v) => !v)}
        aria-expanded={expanded}
      >
        <span>{showLabel}</span>
      </button>
      {expanded && (
        <pre className="aster-tool-card__body">
          {JSON.stringify(message.toolResult, null, 2)}
        </pre>
      )}
    </div>
  );
}

// messages.map 内：
return (
  <div className="aster-messages" ref={scrollRef} onScroll={handleScroll}>
    {messages.map((m) => {
      if (m.role === 'tool') {
        return <ToolResultCard key={m.id} message={m} />;
      }
      return (
        <ChatBubble key={m.id} message={m}
          onRetry={() => void retryMessage(m.id)} onSettings={onSettings} />
      );
    })}
  </div>
);
```
