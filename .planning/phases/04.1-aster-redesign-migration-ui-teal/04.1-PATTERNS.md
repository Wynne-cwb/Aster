```tsx
function formatTime(ts: number): string {
  if (!ts) return '';
  const d = new Date(ts);
  const mo = String(d.getMonth() + 1).padStart(2, '0');
  const da = String(d.getDate()).padStart(2, '0');
  const h = String(d.getHours()).padStart(2, '0');
  const m = String(d.getMinutes()).padStart(2, '0');
  return `${mo}-${da} ${h}:${m}`;
}

// user 气泡
return (
  <div className="msg msg-user">
    <div className="bubble bubble-user">{message.content}</div>
    {message.ts && <span className="msg-time">{formatTime(message.ts)}</span>}
  </div>

    </div>

      <span className="msg-time">{formatTime(message.ts)}</span>
```

## Shared Patterns

### 主题机制（不动）
**Source:** `src/main.tsx`（已有，不改）
**Pattern:** `#root.dataset.theme = "light" | "dark"`，CSS `[data-theme="light"]` / `[data-theme="dark"]` 选择器继承 token
**Apply to:** 所有组件（通过 CSS 变量自动继承，组件无需感知主题）

### focus-visible Ring（全局升级）
**Source:** 旧 styles.css 中 `outline: 2px solid var(--brand)` → 新 `box-shadow: var(--ring-focus)`
**新值:** `var(--ring-focus) = 0 0 0 2px var(--bg), 0 0 0 4px var(--accent)`（双层 ring）
**Apply to:** 所有 `<button>`:focus-visible、`<input>`:focus-visible、`<textarea>`:focus-visible

### Lingui i18n 标记（必须保留）
**Source:** 所有现有组件（`<Trans>`, `useLingui().t`)
**Apply to:** 重皮时所有新增/修改的中文字符串必须沿用 `<Trans>` 或 `t\`\`` macro 包裹
**Anti-pattern:** 直接复制 proto-app.jsx 裸字符串字面量进 JSX——这会导致 lingui extract 警告

### 诚实禁用（贯穿所有组件）
**Source:** 现有约定（CLAUDE.md §UI）
**Pattern:** `aria-disabled="true"` + `opacity: 0.38` + `cursor: not-allowed`，不用 `disabled` HTML 属性（aria-disabled 更灵活）
**Apply to:** PaperclipIcon 附件按钮（InputBar）、give-up 红卡不加假撤销按钮（ChatStream CIRCUIT_OPEN）

### btn-icon 图标按钮样式（新 token 名）
**Source:** `.planning/design/aster-redesign/src/aster.css`（设计包）
**CSS pattern:**
```css
.btn-icon {
  width: 28px; height: 28px;
  display: inline-flex; align-items: center; justify-content: center;
  background: transparent; border: 0; border-radius: var(--radius-2);
```
