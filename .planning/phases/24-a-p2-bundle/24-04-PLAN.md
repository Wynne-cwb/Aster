**B. 创建 src/components/SlidePreviewPanel.tsx**

要求（按 UI-SPEC.md 契约）：
- `export default function SlidePreviewPanel` — default export（供 React.lazy）
- Props: `{ shapes: ShapeSpec[]; }`（containerRef 可选，但工具层通过 registerPreviewElement 注入更简洁）
- 内部 useRef 持有 `.slide-preview-container` DOM ref
- useLayoutEffect 挂载时调 `registerPreviewElement(() => containerRef.current)`，卸载时调 `registerPreviewElement(() => null)`
- useState: `containerWidth`（默认 318px）、`status: 'idle' | 'capturing' | 'error'`
- useRef: `containerElRef`（指向 .slide-preview-container）
- ResizeObserver 监听容器宽度变化，更新 containerWidth（或简单地 onLayout 读 offsetWidth）
- useMemo 调 `mapShapesToRender(shapes, containerWidth)` 缓存渲染结果
- 预览容器 height = `containerWidth * (540 / 960)`（16:9）
- 文案全用 `<Trans>` 宏（Lingui）：「幻灯片预览」「截图自查中…」「自查失败」
- 无障碍：容器 `role="img"` + `aria-label="幻灯片预览"`，标题用 `<span role="heading" aria-level={3}>`
- flex-shrink:0 + min-width:0（CSS 类已含，组件本身无需额外 style）

```typescript
import React, { useLayoutEffect, useMemo, useRef, useState } from 'react';
import { Trans } from '@lingui/react/macro';
import { mapShapesToRender } from '../agent/design/slide-preview';
import { registerPreviewElement } from '../agent/tools/read/visual-check';
import type { ShapeSpec } from '../agent/design/ppt-layouts';

interface Props {
  shapes: ShapeSpec[];
}

export default function SlidePreviewPanel({ shapes }: Props) {
  const containerElRef = useRef<HTMLDivElement>(null);
  const [containerWidth, setContainerWidth] = useState(318); // 默认 318px（350 - 2×16 padding）

  // 挂载时注册 previewEl getter，供 visual_check_slide 工具截图
  useLayoutEffect(() => {
    registerPreviewElement(() => containerElRef.current);
    return () => registerPreviewElement(() => null);
  }, []);

  // 监听容器宽度（ResizeObserver）
  useLayoutEffect(() => {
    const el = containerElRef.current;
    if (!el) return;
    const observer = new ResizeObserver((entries) => {
      const w = entries[0]?.contentRect.width;
      if (w && w > 0) setContainerWidth(w);
    });
    observer.observe(el.parentElement ?? el);
    return () => observer.disconnect();
  }, []);

  const renderedShapes = useMemo(
    () => mapShapesToRender(shapes, containerWidth),
    [shapes, containerWidth],
  );
  const previewHeight = Math.round(containerWidth * (540 / 960));

  return (
    <div className="slide-preview-panel">
      <div className="slide-preview-panel__header">
        <span
          className="slide-preview-panel__title"
          role="heading"
          aria-level={3}
        >
          <Trans>幻灯片预览</Trans>
        </span>
      </div>
      <div
        ref={containerElRef}
        className="slide-preview-container"
        role="img"
        aria-label="幻灯片预览"
        style={{ height: previewHeight }}
      >
        {renderedShapes.map((s) => (
          <div key={s.key} style={s.style as React.CSSProperties}>
            {s.text}
          </div>
        ))}
      </div>
    </div>
  );
}
```

注意：
- `s.style as React.CSSProperties` 是因为 SlideRenderShape.style 是精确字面量类型，React.CSSProperties 兼容接受
- html2canvas 不在面板内 import（截图由工具层 visual_check_slide 触发）
- 颜色物理隔离：面板 chrome 用 CSS 变量，形状内容颜色用 ShapeSpec hex（不用 --accent / --bg 给幻灯片内容着色）

```typescript
// 在 ChatStream 内找最新 apply_slide_layout tool call 的 args
const layoutShapes = useMemo(() => {
  for (let i = messages.length - 1; i >= 0; i--) {
    const msg = messages[i];
    if (msg.role === 'tool' || msg.role === 'assistant') continue;
    // 找最新 assistant 消息中的 apply_slide_layout tool_call
    if (msg.role === 'assistant' && msg.tool_calls) {
      const layoutCall = msg.tool_calls.find((tc) => tc.function?.name === 'apply_slide_layout');
      if (layoutCall) {
        try {
          const parsed = JSON.parse(layoutCall.function.arguments ?? '{}');
          if (Array.isArray(parsed.shapes) && parsed.shapes.length > 0) {
            return parsed.shapes as ShapeSpec[];
          }
        } catch { /* ignore */ }
      }
    }
  }
  return null;
}, [messages]);
```
